import json

def convert(b):
    a = ""
    i = 7
    while i >= 0:
        if b % (2**i) == b:
            a = a + "0"
            i = i - 1
        else:
            a = a + "1"
            b = b - 2**i
            i = i - 1
    print(a)
    x = json.dumps(a)
    print(x)

 
for number in range(10):
